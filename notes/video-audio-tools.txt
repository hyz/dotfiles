### http://savvyadmin.com/extract-audio-from-video-files-to-wav-using-mplayer/
#extract the audio from a video
mplayer -ao pcm:fast:file=audio.wav -vo null -vc null video.avi
#Convert WAV to MP3
lame -V0 -q0 --vbr-new audio.wav audio.mp3
